begin
   decl
    int a[3][3];
    int i,j,s;
enddecl;
    s=0;
    i=0;
    j=0;
    while(i<3)do
        while(j<3)do
            read(a[i][j]);
            s = a[i][j]+s;
            j=j+1;
        endwhile;
        j=0;
        i=i+1;
    endwhile;
    write(a[0][0]);
    write(a[1][1]);
    write(a[2][2]);
end;
